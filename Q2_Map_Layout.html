from qgis.core import *
from qgis.gui import *
from PyQt5.QtGui import QColor, QFont, QImage, QPainter
from PyQt5.QtCore import QVariant, QRectF, QSizeF, Qt
import os

project = QgsProject.instance()

areas_list      = project.mapLayersByName("Areas")
gardens_list    = project.mapLayersByName("Gardens")
highways_list   = project.mapLayersByName("Highways")
railways_list   = project.mapLayersByName("Railway_Tracks")
stations_list   = project.mapLayersByName("Railway_Stations")
restaurants_list = project.mapLayersByName("Restaurants")
hospitals_list  = project.mapLayersByName("Hospitals")
buildings_list  = project.mapLayersByName("Buildings")

required = [areas_list, gardens_list, highways_list, railways_list,
            stations_list, restaurants_list, hospitals_list, buildings_list]

if any(not layer_list for layer_list in required):
    print("ERROR: Please run Q1_Create_Map.py first to create all layers!")
    raise Exception("Required layers not found. Run Q1 first.")

areas       = areas_list[0]
gardens     = gardens_list[0]
highways    = highways_list[0]
railways    = railways_list[0]
stations    = stations_list[0]
restaurants = restaurants_list[0]
hospitals   = hospitals_list[0]
buildings   = buildings_list[0]

print("=" * 65)
print("  Q2: Creating Map Layout with Cartographic Elements")
print("=" * 65)

print("\n  [Q2.3] Applying Hospital symbol (Red Cross)...")

svg_symbol = QgsMarkerSymbol()
sym_layer = QgsSimpleMarkerSymbolLayer()
sym_layer.setShape(QgsSimpleMarkerSymbolLayer.Cross)
sym_layer.setColor(QColor(255, 0, 0))
sym_layer.setStrokeColor(QColor(200, 0, 0))
sym_layer.setSize(5.0)
sym_layer.setStrokeWidth(0.8)
svg_symbol.changeSymbolLayer(0, sym_layer)
hospitals.renderer().setSymbol(svg_symbol)
hospitals.triggerRepaint()
print("  -> Hospitals now displayed with RED CROSS symbol")

print("\n  [Q2.4] Applying Metro symbol to Andheri station...")

categories = []

metro_symbol = QgsMarkerSymbol()
metro_layer = QgsSimpleMarkerSymbolLayer()
metro_layer.setShape(QgsSimpleMarkerSymbolLayer.Circle)
metro_layer.setColor(QColor(0, 100, 200))
metro_layer.setStrokeColor(QColor(0, 50, 150))
metro_layer.setSize(6.0)
metro_layer.setStrokeWidth(0.5)
metro_symbol.changeSymbolLayer(0, metro_layer)

font_layer = QgsFontMarkerSymbolLayer()
font_layer.setFontFamily("Arial")
font_layer.setCharacter("M")
font_layer.setSize(3.5)
font_layer.setColor(QColor(255, 255, 255))
metro_symbol.appendSymbolLayer(font_layer)

cat_metro = QgsRendererCategory("Andheri", metro_symbol, "Metro Station (Andheri)")
categories.append(cat_metro)

for feat in stations.getFeatures():
    sname = feat['name']
    if sname != "Andheri":
        default_symbol = QgsMarkerSymbol.createSimple({
            'name': 'square',
            'color': '30,30,30',
            'size': '3'
        })
        cat = QgsRendererCategory(sname, default_symbol, sname)
        categories.append(cat)

categorized_renderer = QgsCategorizedSymbolRenderer("name", categories)
stations.setRenderer(categorized_renderer)
stations.triggerRepaint()
print("  -> Andheri station now has METRO symbol (Blue circle with M)")

iface.mapCanvas().refresh()

print("\n  Creating Print Layout...")

manager = project.layoutManager()
existing = manager.layoutByName("GIS_Practical_Map")
if existing:
    manager.removeLayout(existing)

layout = QgsPrintLayout(project)
layout.initializeDefaults()
layout.setName("GIS_Practical_Map")

page = layout.pageCollection().page(0)
page.setPageSize(QgsLayoutSize(420, 297, QgsUnitTypes.LayoutMillimeters))

map_item = QgsLayoutItemMap(layout)
map_item.setRect(QRectF(0, 0, 370, 250))
map_item.attemptMove(QgsLayoutPoint(10, 35, QgsUnitTypes.LayoutMillimeters))
map_item.attemptResize(QgsLayoutSize(370, 250, QgsUnitTypes.LayoutMillimeters))
map_item.setExtent(areas.extent())
map_item.setFrameEnabled(True)
map_item.setFrameStrokeWidth(QgsLayoutMeasurement(0.5, QgsUnitTypes.LayoutMillimeters))
layout.addLayoutItem(map_item)

title = QgsLayoutItemLabel(layout)
title.setText("Residential Area Map - Mumbai (Andheri-Goregaon)")
title.setFont(QFont("Arial", 18, QFont.Bold))
title.adjustSizeToText()
title.attemptMove(QgsLayoutPoint(10, 5, QgsUnitTypes.LayoutMillimeters))
title.attemptResize(QgsLayoutSize(370, 15, QgsUnitTypes.LayoutMillimeters))
title.setHAlign(Qt.AlignCenter)
layout.addLayoutItem(title)

subtitle = QgsLayoutItemLabel(layout)
subtitle.setText("T.Y. B.Sc. IT - Fundamentals of GIS | University of Mumbai")
subtitle.setFont(QFont("Arial", 10))
subtitle.adjustSizeToText()
subtitle.attemptMove(QgsLayoutPoint(10, 22, QgsUnitTypes.LayoutMillimeters))
subtitle.attemptResize(QgsLayoutSize(370, 10, QgsUnitTypes.LayoutMillimeters))
subtitle.setHAlign(Qt.AlignCenter)
layout.addLayoutItem(subtitle)

print("  [Q2.1] Adding North Arrow...")

north_arrow = QgsLayoutItemPicture(layout)
north_arrow.attemptMove(QgsLayoutPoint(385, 35, QgsUnitTypes.LayoutMillimeters))
north_arrow.attemptResize(QgsLayoutSize(25, 40, QgsUnitTypes.LayoutMillimeters))

import glob
qgis_path = QgsApplication.pkgDataPath()
svg_paths = glob.glob(os.path.join(qgis_path, "svg", "arrows", "NorthArrow_*.svg"))
if svg_paths:
    north_arrow.setPicturePath(svg_paths[0])
    print(f"  -> North Arrow SVG: {os.path.basename(svg_paths[0])}")
else:

    svg_paths2 = glob.glob(os.path.join(qgis_path, "svg", "**", "north*"), recursive=True)
    if svg_paths2:
        north_arrow.setPicturePath(svg_paths2[0])
    else:

        north_arrow.setNorthMode(QgsLayoutItemPicture.GridNorth)

north_arrow.setFrameEnabled(True)
layout.addLayoutItem(north_arrow)
print("  -> North Arrow added to layout")

print("  [Q2.2] Adding Legend...")

legend = QgsLayoutItemLegend(layout)
legend.setLinkedMap(map_item)
legend.setTitle("Map Legend")
legend.setFrameEnabled(True)
legend.setBackgroundEnabled(True)
legend.setBackgroundColor(QColor(255, 255, 255, 220))
legend.attemptMove(QgsLayoutPoint(385, 80, QgsUnitTypes.LayoutMillimeters))
legend.attemptResize(QgsLayoutSize(30, 80, QgsUnitTypes.LayoutMillimeters))

legend.setAutoUpdateModel(True)

layout.addLayoutItem(legend)
print("  -> Legend added to layout")

print("  [Q2.2] Adding Scale Bar...")

scalebar = QgsLayoutItemScaleBar(layout)
scalebar.setLinkedMap(map_item)
scalebar.setStyle('Single Box')
scalebar.setUnits(QgsUnitTypes.DistanceKilometers)
scalebar.setNumberOfSegments(4)
scalebar.setNumberOfSegmentsLeft(0)
scalebar.setUnitsPerSegment(0.5)
scalebar.setFont(QFont("Arial", 8))
scalebar.setFrameEnabled(False)
scalebar.attemptMove(QgsLayoutPoint(15, 270, QgsUnitTypes.LayoutMillimeters))
layout.addLayoutItem(scalebar)
print("  -> Scale Bar added to layout")

print("\n  [Q2.5] Exporting map to JPEG at 1500 DPI...")

exporter = QgsLayoutExporter(layout)
settings = QgsLayoutExporter.ImageExportSettings()
settings.dpi = 1500

download_dir = os.path.join(os.path.expanduser("~"), "Downloads")
output_path = os.path.join(download_dir, "Residential_Area_Map_1500dpi.jpg")

result = exporter.exportToImage(output_path, settings)

if result == QgsLayoutExporter.Success:
    print(f"  -> Map exported successfully!")
    print(f"  -> File: {output_path}")
    print(f"  -> Resolution: 1500 DPI")
    print(f"  -> Format: JPEG")
else:
    print(f"  -> Export result code: {result}")
    print("  -> You can manually export from: Layout Manager > Export as Image")

manager.addLayout(layout)
print("  -> Layout 'GIS_Practical_Map' added to Layout Manager")

print("\n" + "=" * 65)
print("  Q2: MAP LAYOUT CREATED SUCCESSFULLY!")
print("=" * 65)
print(\
   )
print("=" * 65)
