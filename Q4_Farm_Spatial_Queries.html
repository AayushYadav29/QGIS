from qgis.core import *
from qgis.gui import *
from PyQt5.QtGui import QColor
from PyQt5.QtCore import QVariant

project = QgsProject.instance()

farms_list    = project.mapLayersByName("Farms")
wells_list    = project.mapLayersByName("Wells")
paths_list    = project.mapLayersByName("Paths")
highway_list  = project.mapLayersByName("Highway")

required = [farms_list, wells_list, paths_list, highway_list]

if any(not layer_list for layer_list in required):
    print("ERROR: Please run Q3_Create_Farm_Map.py first to create all layers!")
    raise Exception("Required layers not found. Run Q3_Create_Farm_Map.py first.")

farms   = farms_list[0]
wells   = wells_list[0]
paths   = paths_list[0]
highway = highway_list[0]

print("=" * 65)
print("  Q2: Spatial Queries on Farm Map - RESULTS")
print("=" * 65)

print("\n" + "-" * 65)
print("  Q2.1: Farms That Contain a Well")
print("-" * 65)

farm_features = []
for feat in farms.getFeatures():
    farm_features.append(feat)

farms_with_wells = set()
for well_feat in wells.getFeatures():
    well_geom = well_feat.geometry()
    for farm_feat in farm_features:
        farm_geom = farm_feat.geometry()
        if farm_geom.contains(well_geom):
            farms_with_wells.add(farm_feat.id())
            print(f"  FOUND: Well '{well_feat['location']}' (ID: {well_feat['id']}) "
                  f"is inside Farm '{farm_feat['own_name']}' (ID: {farm_feat['id']})")

if not farms_with_wells:
    print("  No farms contain wells.")
else:

    farms.removeSelection()
    farms.select(list(farms_with_wells))
    iface.setActiveLayer(farms)
    iface.mapCanvas().refresh()
    print(f"\n  [Total: {len(farms_with_wells)} farm(s) contain wells]")
    print("  [Farms with wells are now SELECTED/HIGHLIGHTED on the map]")

print("\n" + "-" * 65)
print("  Q2.2: Calculate Area of Each Farm & Show Largest Farm")
print("-" * 65)

field_names = [field.name() for field in farms.fields()]
if "Farm_area" not in field_names:
    farms.startEditing()
    farms.dataProvider().addAttributes([QgsField("Farm_area", QVariant.Double)])
    farms.updateFields()
    farms.commitChanges()

d = QgsDistanceArea()
d.setEllipsoid('WGS84')
d.setSourceCrs(farms.crs(), project.transformContext())

farms.startEditing()
area_idx = farms.fields().indexOf("Farm_area")

max_area = 0
largest_farm_name = ""
largest_farm_id = None

for feat in farms.getFeatures():
    area = d.measureArea(feat.geometry())
    area_sq_m = d.convertAreaMeasurement(area, QgsUnitTypes.AreaSquareMeters)
    farms.changeAttributeValue(feat.id(), area_idx, round(area_sq_m, 2))
    print(f"  Farm ID {feat['id']}: {feat['own_name']:20s} -> Area = {round(area_sq_m, 2):>12.2f} sq meters")

    if area_sq_m > max_area:
        max_area = area_sq_m
        largest_farm_name = feat['own_name']
        largest_farm_id = feat.id()

farms.commitChanges()
print(f"\n  [Farm_area column added and populated successfully]")

print(f"\n  ** LARGEST FARM: {largest_farm_name} **")
print(f"     Area: {max_area:.2f} sq meters")

farms.removeSelection()
farms.select(largest_farm_id)
iface.setActiveLayer(farms)
iface.mapCanvas().refresh()
print("  [Largest farm is now SELECTED/HIGHLIGHTED on the map]")

print("\n" + "-" * 65)
print("  Q2.3: Paths That Intersect with Highway")
print("-" * 65)

highway_geoms = []
for feat in highway.getFeatures():
    highway_geoms.append((feat.geometry(), feat['id']))

intersecting_path_ids = []
for path_feat in paths.getFeatures():
    path_geom = path_feat.geometry()
    for hw_geom, hw_id in highway_geoms:
        if path_geom.intersects(hw_geom):
            intersecting_path_ids.append(path_feat.id())
            print(f"  INTERSECTS: Path ID {path_feat['id']} intersects Highway '{hw_id}'")

if not intersecting_path_ids:
    print("  No paths intersect with the highway.")
else:

    paths.removeSelection()
    paths.select(intersecting_path_ids)
    iface.setActiveLayer(paths)
    iface.mapCanvas().refresh()
    print(f"\n  [Total: {len(intersecting_path_ids)} path(s) intersect with highway]")
    print("  [Intersecting paths are now SELECTED/HIGHLIGHTED on the map]")

print("\n" + "=" * 65)
print("  Q2: All Spatial Queries Executed Successfully!")
print("=" * 65)
