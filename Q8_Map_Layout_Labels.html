from qgis.core import *
from qgis.gui import *
from PyQt5.QtGui import QColor, QFont
from PyQt5.QtCore import QVariant, QRectF, QSizeF
import os

project = QgsProject.instance()
download_dir = os.path.join(os.path.expanduser("~"), "Downloads")

colleges_list    = project.mapLayersByName("Colleges")
restaurants_list = project.mapLayersByName("Restaurants")
hospitals_list   = project.mapLayersByName("Hospitals")
roads_list       = project.mapLayersByName("Roads")
railways_list    = project.mapLayersByName("Railway_Tracks")

required = [colleges_list, restaurants_list, hospitals_list, roads_list, railways_list]

if any(not l for l in required):
    print("ERROR: Please run Q11_Vector_Points_Map.py first!")
    raise Exception("Required layers not found. Run Q11 first.")

colleges    = colleges_list[0]
restaurants = restaurants_list[0]
hospitals   = hospitals_list[0]
roads       = roads_list[0]
railways    = railways_list[0]

all_layers = [colleges, restaurants, hospitals, roads, railways]

print("=" * 65)
print("  Q2: Creating Map Layout with Labels, Legend, Scale & Title")
print("=" * 65)

print("\n  [Q2.1] Adding/Verifying Labels on all layers...")

label_configs = [
    (colleges, "name"),
    (restaurants, "name"),
    (hospitals, "name"),
    (roads, "name"),
    (railways, "name")
]

for layer, field in label_configs:
    settings = QgsPalLayerSettings()
    settings.fieldName = field
    settings.enabled = True

    text_format = QgsTextFormat()
    text_format.setSize(8)
    text_format.setColor(QColor(0, 0, 0))

    buf = QgsTextBufferSettings()
    buf.setEnabled(True)
    buf.setSize(1.2)
    buf.setColor(QColor(255, 255, 255))
    text_format.setBuffer(buf)

    settings.setFormat(text_format)
    labeling = QgsVectorLayerSimpleLabeling(settings)
    layer.setLabelsEnabled(True)
    layer.setLabeling(labeling)
    layer.triggerRepaint()
    print(f"    Labels ON: {layer.name()} (field: '{field}')")

iface.mapCanvas().refresh()

layout_name = "LocationMap_Layout"

manager = project.layoutManager()
existing = manager.layoutByName(layout_name)
if existing:
    manager.removeLayout(existing)

layout = QgsPrintLayout(project)
layout.initializeDefaults()
layout.setName(layout_name)
manager.addLayout(layout)

map_item = QgsLayoutItemMap(layout)
map_item.setRect(20, 20, 200, 250)
map_item.setExtent(iface.mapCanvas().extent())
map_item.setCrs(QgsCoordinateReferenceSystem("EPSG:4326"))
layout.addLayoutItem(map_item)
map_item.attemptMove(QgsLayoutPoint(15, 35, QgsUnitTypes.LayoutMillimeters))
map_item.attemptResize(QgsLayoutSize(175, 210, QgsUnitTypes.LayoutMillimeters))

print("\n  Map item added to layout")

print("\n  [Q2.4] Adding Title with proper formatting...")

title = QgsLayoutItemLabel(layout)
title.setText("Mumbai - Colleges, Restaurants & Hospitals Map")
title.setFont(QFont("Arial", 18, QFont.Bold))
title.setFontColor(QColor(0, 50, 120))
title.setHAlign(Qt.AlignCenter)
title.adjustSizeToText()
layout.addLayoutItem(title)
title.attemptMove(QgsLayoutPoint(15, 5, QgsUnitTypes.LayoutMillimeters))
title.attemptResize(QgsLayoutSize(175, 15, QgsUnitTypes.LayoutMillimeters))
print("    Title: 'Mumbai - Colleges, Restaurants & Hospitals Map'")
print("    Font: Arial 18pt Bold, Color: Dark Blue")

subtitle = QgsLayoutItemLabel(layout)
subtitle.setText("GIS Practical Examination | CRS: WGS 84 (EPSG:4326)")
subtitle.setFont(QFont("Arial", 9))
subtitle.setFontColor(QColor(80, 80, 80))
subtitle.setHAlign(Qt.AlignCenter)
subtitle.adjustSizeToText()
layout.addLayoutItem(subtitle)
subtitle.attemptMove(QgsLayoutPoint(15, 22, QgsUnitTypes.LayoutMillimeters))
subtitle.attemptResize(QgsLayoutSize(175, 10, QgsUnitTypes.LayoutMillimeters))
print("    Subtitle added with CRS info")

print("\n  [Q2.2] Adding Legend...")

legend = QgsLayoutItemLegend(layout)
legend.setTitle("Map Legend")
legend.setLinkedMap(map_item)
legend.setStyleFont(QgsLegendStyle.Title, QFont("Arial", 11, QFont.Bold))
legend.setStyleFont(QgsLegendStyle.Subgroup, QFont("Arial", 9, QFont.Bold))
legend.setStyleFont(QgsLegendStyle.SymbolLabel, QFont("Arial", 8))
layout.addLayoutItem(legend)
legend.attemptMove(QgsLayoutPoint(195, 35, QgsUnitTypes.LayoutMillimeters))
print("    Legend added with styled fonts")

print("\n  [Q2.3] Adding Scale Bar...")

scalebar = QgsLayoutItemScaleBar(layout)
scalebar.setStyle('Single Box')
scalebar.setLinkedMap(map_item)
scalebar.setMapUnitsPerScaleBarUnit(1000)
scalebar.setNumberOfSegments(4)
scalebar.setNumberOfSegmentsLeft(0)
scalebar.setUnitsPerSegment(1)
scalebar.setUnitLabel("km")
scalebar.setFont(QFont("Arial", 8))
layout.addLayoutItem(scalebar)
scalebar.attemptMove(QgsLayoutPoint(15, 250, QgsUnitTypes.LayoutMillimeters))
print("    Scale Bar: Single Box, 4 segments, km units")

north = QgsLayoutItemLabel(layout)
north.setText("N\n▲")
north.setFont(QFont("Arial", 16, QFont.Bold))
north.setFontColor(QColor(0, 0, 0))
north.setHAlign(Qt.AlignCenter)
north.adjustSizeToText()
layout.addLayoutItem(north)
north.attemptMove(QgsLayoutPoint(195, 220, QgsUnitTypes.LayoutMillimeters))
print("    North Arrow added")

footer = QgsLayoutItemLabel(layout)
footer.setText("Prepared for: T.Y. B.Sc. IT - Semester VI | University of Mumbai")
footer.setFont(QFont("Arial", 7))
footer.setFontColor(QColor(120, 120, 120))
footer.setHAlign(Qt.AlignCenter)
footer.adjustSizeToText()
layout.addLayoutItem(footer)
footer.attemptMove(QgsLayoutPoint(15, 265, QgsUnitTypes.LayoutMillimeters))
footer.attemptResize(QgsLayoutSize(175, 8, QgsUnitTypes.LayoutMillimeters))

output_path = os.path.join(download_dir, "Location_Map_Output.jpg")

exporter = QgsLayoutExporter(layout)
settings = QgsLayoutExporter.ImageExportSettings()
settings.dpi = 300

result = exporter.exportToImage(output_path, settings)

if result == QgsLayoutExporter.Success:
    print(f"\n  Map exported: Location_Map_Output.jpg (300 dpi)")
else:
    print(f"\n  Export status code: {result}")

print("\n" + "=" * 65)
print("  Q2: Map Layout - COMPLETED SUCCESSFULLY!")
print("=" * 65)
print("  1. Labels   : Enabled on all 5 layers (name field)")
print("  2. Legend    : Added with styled fonts (Title, Groups, Symbols)")
print("  3. Scale Bar : Single Box, 4 segments, km units")
print("  4. Title     : 'Mumbai - Colleges, Restaurants & Hospitals Map'")
print("                 Arial 18pt Bold, Dark Blue, centered")
print(f"\n  Output: Location_Map_Output.jpg")
print(f"  Layout: {layout_name}")
print("  Open via: Project > Layouts > LocationMap_Layout")
print("=" * 65)
