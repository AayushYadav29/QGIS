# ==================================================================
#  STEP 5: Calculate Line Lengths for Railways
# ==================================================================
print("\n" + "-" * 70)
print("  STEP 5: Calculating Railway Line Lengths")
print("-" * 70)

# Add "Length_km" field
rails_layer.startEditing()
rails_layer.dataProvider().addAttributes([QgsField("Length_km", QVariant.Double)])
rails_layer.updateFields()
rails_layer.commitChanges()

# Setup ellipsoidal distance calculator
d = QgsDistanceArea()
d.setEllipsoid('WGS84')
d.setSourceCrs(rails_layer.crs(), project.transformContext())

rails_layer.startEditing()
length_idx = rails_layer.fields().indexOf("Length_km")

lengths = []
for feat in rails_layer.getFeatures():
    geom = feat.geometry()
    length_m  = d.measureLength(geom)
    length_km = d.convertLengthMeasurement(length_m, QgsUnitTypes.DistanceKilometers)
    rails_layer.changeAttributeValue(feat.id(), length_idx, round(length_km, 4))
    lengths.append(length_km)
    print(f"    {feat['Name']:35s} -> Length = {length_km:>10.4f} km")

rails_layer.commitChanges()

# ==================================================================
#  STEP 6: Railway Line Length Statistics
# ==================================================================
print("\n" + "-" * 70)
print("  STEP 6: Railway Line Length Statistics")
print("-" * 70)

total_length = sum(lengths)
avg_length   = total_length / len(lengths)
min_length   = min(lengths)
max_length   = max(lengths)

print(f"\n    Total Railway Features  : {len(lengths)}")
print(f"    Total Length            : {total_length:>12.4f} km")
print(f"    Average Length          : {avg_length:>12.4f} km")
print(f"    Minimum Length          : {min_length:>12.4f} km")
print(f"    Maximum Length          : {max_length:>12.4f} km")

# ==================================================================
#  STEP 7: Calculate Area of India (Admin Boundary)
# ==================================================================
print("\n" + "-" * 70)
print("  STEP 7: Calculate Area of India (Admin Boundary)")
print("-" * 70)

adm0_layer.startEditing()
adm0_layer.dataProvider().addAttributes([QgsField("Area_sqkm", QVariant.Double)])
adm0_layer.updateFields()
adm0_layer.commitChanges()

d_area = QgsDistanceArea()
d_area.setEllipsoid('WGS84')
d_area.setSourceCrs(adm0_layer.crs(), project.transformContext())

adm0_layer.startEditing()
area_idx = adm0_layer.fields().indexOf("Area_sqkm")

total_india_area = 0
for feat in adm0_layer.getFeatures():
    area_m2  = d_area.measureArea(feat.geometry())
    area_km2 = d_area.convertAreaMeasurement(area_m2, QgsUnitTypes.AreaSquareKilometers)
    adm0_layer.changeAttributeValue(feat.id(), area_idx, round(area_km2, 4))
    total_india_area += area_km2
    print(f"    {feat['NAME_0']:30s} -> Area = {area_km2:>14.4f} sq km")

adm0_layer.commitChanges()
print(f"\n    Total India Area : {total_india_area:>14.4f} sq km")

# ==================================================================
#  STEP 8: Calculate Total Area Covered by Water Bodies
# ==================================================================
print("\n" + "-" * 70)
print("  STEP 8: Total Area Covered by Water Bodies in India")
print("-" * 70)

water_layer.startEditing()
water_layer.dataProvider().addAttributes([QgsField("Area_sqkm", QVariant.Double)])
water_layer.updateFields()
water_layer.commitChanges()

d_water = QgsDistanceArea()
d_water.setEllipsoid('WGS84')
d_water.setSourceCrs(water_layer.crs(), project.transformContext())

water_layer.startEditing()
w_area_idx = water_layer.fields().indexOf("Area_sqkm")

total_water_area = 0
for feat in water_layer.getFeatures():
    area_m2  = d_water.measureArea(feat.geometry())
    area_km2 = d_water.convertAreaMeasurement(area_m2, QgsUnitTypes.AreaSquareKilometers)
    water_layer.changeAttributeValue(feat.id(), w_area_idx, round(area_km2, 4))
    total_water_area += area_km2
    print(f"    {feat['Name']:25s} ({feat['Type']:10s}) -> Area = {area_km2:>10.4f} sq km")

water_layer.commitChanges()

pct = (total_water_area / total_india_area) * 100 if total_india_area > 0 else 0

print(f"\n    Total Water Bodies        : {water_layer.featureCount()}")
print(f"    Total Water Body Area     : {total_water_area:>12.4f} sq km")
print(f"    Total India Area          : {total_india_area:>12.4f} sq km")
print(f"    Water Coverage of India   : {pct:>12.4f} %")
