from qgis.core import *
from qgis.gui import *
from qgis.utils import iface, plugins, available_plugins, active_plugins
from pyplugin_installer import instance as plugin_installer_instance
from pyplugin_installer.installer_data import repositories, plugins as plugin_data

print("=" * 65)
print("  Q2: Install and Enable Plugins (Experimental)")
print("=" * 65)

print("\n" + "-" * 65)
print("  STEP 1: Enable Experimental Plugins")
print("-" * 65)

from qgis.PyQt.QtCore import QSettings

settings = QSettings()

settings.setValue("app/plugin_installer/allowExperimental", True)
print("  [OK] Experimental plugins: ENABLED")

settings.setValue("app/plugin_installer/allowDeprecated", False)
print("  [OK] Deprecated plugins: DISABLED (best practice)")

settings.setValue("app/plugin_installer/checkOnStart", True)
print("  [OK] Check for updates on startup: ENABLED")

print("\n" + "-" * 65)
print("  STEP 2: Refresh Plugin Repository")
print("-" * 65)

try:
    installer = plugin_installer_instance()
    installer.fetchAvailablePlugins(False)
    print("  [OK] Plugin repository refreshed successfully")
except Exception as e:
    print(f"  [NOTE] Repository refresh: {e}")
    print("  (This is normal if run outside the main QGIS window)")

print("\n" + "-" * 65)
print("  STEP 3: Install Plugins")
print("-" * 65)

plugins_to_install = [
    ("QuickMapServices",     "Adds basemaps (Google, OSM, Bing, etc.)"),
    ("qgis2web",             "Export QGIS project to web map (OpenLayers/Leaflet)"),
    ("QuickOSM",             "Download OpenStreetMap data directly"),
    ("Semi-Automatic Classification Plugin", "Remote sensing & land classification"),
]

for plugin_name, description in plugins_to_install:
    print(f"\n  Plugin: {plugin_name}")
    print(f"  Purpose: {description}")

    try:

        if plugin_name in active_plugins:
            print(f"  Status: Already installed and ACTIVE")
            continue

        installer = plugin_installer_instance()
        installer.installPlugin(plugin_name)
        print(f"  Status: INSTALLED successfully")

    except Exception as e:
        print(f"  Status: Could not install -> {e}")
        print(f"  Manual: Plugins > Manage and Install Plugins > Search '{plugin_name}'")

print("\n\n" + "-" * 65)
print("  STEP 4: Enable/Activate Plugins")
print("-" * 65)

for plugin_name, description in plugins_to_install:
    try:

        if plugin_name not in active_plugins:
            iface.pluginManagerInterface().loadPlugin(plugin_name)
            iface.pluginManagerInterface().startPlugin(plugin_name)
            print(f"  [OK] {plugin_name}: ENABLED")
        else:
            print(f"  [OK] {plugin_name}: Already ACTIVE")
    except Exception as e:
        print(f"  [--] {plugin_name}: {e}")

print("\n\n" + "-" * 65)
print("  STEP 5: All Installed Plugins")
print("-" * 65)

print(f"\n  {'Plugin Name':<45s} {'Status':<10s}")
print(f"  {'='*45} {'='*10}")

try:
    all_plugins = available_plugins
    active = active_plugins

    for pname in sorted(all_plugins):
        status = "ACTIVE" if pname in active else "INACTIVE"
        print(f"  {pname:<45s} {status:<10s}")

    print(f"\n  Total plugins available: {len(all_plugins)}")
    print(f"  Active plugins: {len(active)}")
except Exception as e:
    print(f"  [NOTE] Could not list plugins: {e}")

print("\n\n" + "-" * 65)
print("  MANUAL STEPS (for reference in exam):")
print("-" * 65)
print(\
   )

print("=" * 65)
print("  Q2: Plugin Management - COMPLETED!")
print("=" * 65)
print("  1. Experimental plugins ENABLED in settings")
print("  2. Plugin repository refreshed")
print("  3. Attempted installation of useful plugins:")
for pname, desc in plugins_to_install:
    print(f"     - {pname}")
print("  4. All plugins activated")
print("  5. Full plugin list displayed")
print("=" * 65)
