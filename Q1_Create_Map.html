from qgis.core import *

project = QgsProject.instance()
project.setTitle("GIS Practical - Residential Area Map (Mumbai)")

area_layer = QgsVectorLayer("Polygon?crs=EPSG:4326", "Areas", "memory")
area_provider = area_layer.dataProvider()
area_provider.addAttributes([
    QgsField("id", QVariant.Int),
    QgsField("name", QVariant.String)
])
area_layer.updateFields()

areas_data = [
    (1, "Andheri East",   [(72.855,19.13),(72.88,19.13),(72.88,19.16),(72.855,19.16),(72.855,19.13)]),
    (2, "Andheri West",   [(72.83,19.13),(72.855,19.13),(72.855,19.16),(72.83,19.16),(72.83,19.13)]),
    (3, "Goregaon East",  [(72.855,19.10),(72.88,19.10),(72.88,19.13),(72.855,19.13),(72.855,19.10)]),
    (4, "Goregaon West",  [(72.83,19.10),(72.855,19.10),(72.855,19.13),(72.83,19.13),(72.83,19.10)])
]

for aid, name, coords in areas_data:
    feat = QgsFeature()
    points = [QgsPointXY(x, y) for x, y in coords]
    feat.setGeometry(QgsGeometry.fromPolygonXY([points]))
    feat.setAttributes([aid, name])
    area_provider.addFeature(feat)

area_layer.updateExtents()
project.addMapLayer(area_layer)

symbol = QgsFillSymbol.createSimple({
    'color': '180,200,255,60',
    'outline_color': '0,0,180',
    'outline_width': '0.5'
})
area_layer.renderer().setSymbol(symbol)
area_layer.triggerRepaint()

    area_layer, garden_layer, highway_layer, railway_layer,
    station_layer, restaurant_layer, hospital_layer, building_layer
]

for layer in all_layers:
    settings = QgsPalLayerSettings()
    settings.fieldName = "name"
    settings.enabled = True
    text_format = QgsTextFormat()
    text_format.setSize(8)
    buffer = QgsTextBufferSettings()
    buffer.setEnabled(True)
    buffer.setSize(1)
    buffer.setColor(QColor(255, 255, 255))
    text_format.setBuffer(buffer)
    settings.setFormat(text_format)
    labeling = QgsVectorLayerSimpleLabeling(settings)
    layer.setLabelsEnabled(True)
    layer.setLabeling(labeling)
    layer.triggerRepaint()
