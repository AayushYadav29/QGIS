from qgis.core import *
from qgis.gui import *
from PyQt5.QtGui import QColor
from PyQt5.QtCore import QVariant

project = QgsProject.instance()

gardens_list    = project.mapLayersByName("Gardens")
highways_list   = project.mapLayersByName("Highways")
railways_list   = project.mapLayersByName("Railway_Tracks")
restaurants_list = project.mapLayersByName("Restaurants")

if not gardens_list or not highways_list or not railways_list or not restaurants_list:
    print("ERROR: Please run Q1_Create_Map.py first to create all layers!")
    raise Exception("Required layers not found. Run Q1 first.")

gardens     = gardens_list[0]
highways    = highways_list[0]
railways    = railways_list[0]
restaurants = restaurants_list[0]

print("=" * 65)
print("  Q2: Spatial Queries - RESULTS")
print("=" * 65)

print("\n" + "-" * 65)
print("  Q2.1: Add Garden_area column & Calculate Area of Each Garden")
print("-" * 65)

field_names = [field.name() for field in gardens.fields()]
if "Garden_area" not in field_names:
    gardens.startEditing()
    gardens.dataProvider().addAttributes([QgsField("Garden_area", QVariant.Double)])
    gardens.updateFields()
    gardens.commitChanges()

d = QgsDistanceArea()
d.setEllipsoid('WGS84')
d.setSourceCrs(gardens.crs(), project.transformContext())

gardens.startEditing()
area_idx = gardens.fields().indexOf("Garden_area")

for feat in gardens.getFeatures():
    area = d.measureArea(feat.geometry())
    area_sq_m = d.convertAreaMeasurement(area, QgsUnitTypes.AreaSquareMeters)
    gardens.changeAttributeValue(feat.id(), area_idx, round(area_sq_m, 2))
    print(f"  Garden ID {feat['id']}: {feat['name']:20s} -> Area = {round(area_sq_m, 2):>12.2f} sq meters")

gardens.commitChanges()
print("  [Garden_area column added and populated successfully]")

print("\n" + "-" * 65)
print("  Q2.2: Show the Largest Garden")
print("-" * 65)

max_area = 0
largest_garden_name = ""
largest_garden_id = None

for feat in gardens.getFeatures():
    if feat['Garden_area'] and feat['Garden_area'] > max_area:
        max_area = feat['Garden_area']
        largest_garden_name = feat['name']
        largest_garden_id = feat.id()

print(f"  LARGEST GARDEN: {largest_garden_name}")
print(f"  Area: {max_area:.2f} sq meters")

gardens.removeSelection()
gardens.select(largest_garden_id)
iface.setActiveLayer(gardens)
iface.mapCanvas().refresh()
print("  [Largest garden is now SELECTED/HIGHLIGHTED on the map]")

print("\n" + "-" * 65)
print("  Q2.3: Railway Tracks that INTERSECT with Highways")
print("-" * 65)

highway_geoms = []
highway_names = []
for feat in highways.getFeatures():
    highway_geoms.append(feat.geometry())
    highway_names.append(feat['name'])

intersecting_track_ids = []
for track_feat in railways.getFeatures():
    track_geom = track_feat.geometry()
    for i, hw_geom in enumerate(highway_geoms):
        if track_geom.intersects(hw_geom):
            intersecting_track_ids.append(track_feat.id())
            print(f"  INTERSECTS: '{track_feat['name']}' intersects '{highway_names[i]}'")

if not intersecting_track_ids:
    print("  No intersecting railway tracks found.")
else:

    railways.removeSelection()
    railways.select(intersecting_track_ids)
    print(f"\n  [Total: {len(set(intersecting_track_ids))} track(s) intersect with highways]")
    print("  [Intersecting tracks are now SELECTED/HIGHLIGHTED on the map]")

print("\n" + "-" * 65)
print("  Q2.4: Restaurants INSIDE Gardens")
print("-" * 65)

garden_data = []
for feat in gardens.getFeatures():
    garden_data.append((feat['name'], feat.geometry()))

restaurant_inside_ids = []
for rest_feat in restaurants.getFeatures():
    rest_geom = rest_feat.geometry()
    for garden_name, garden_geom in garden_data:
        if garden_geom.contains(rest_geom):
            restaurant_inside_ids.append(rest_feat.id())
            print(f"  INSIDE: '{rest_feat['name']}' is inside '{garden_name}'")

if not restaurant_inside_ids:
    print("  No restaurants found inside gardens.")
else:

    restaurants.removeSelection()
    restaurants.select(restaurant_inside_ids)
    print(f"\n  [Total: {len(restaurant_inside_ids)} restaurant(s) found inside gardens]")
    print("  [Restaurants inside gardens are now SELECTED/HIGHLIGHTED on the map]")

print("\n" + "-" * 65)
print("  Q2.5: Railway Tracks that DO NOT Intersect with Highways")
print("-" * 65)

non_intersecting_ids = []
for track_feat in railways.getFeatures():
    track_geom = track_feat.geometry()
    intersects_any = False
    for hw_geom in highway_geoms:
        if track_geom.intersects(hw_geom):
            intersects_any = True
            break
    if not intersects_any:
        non_intersecting_ids.append(track_feat.id())
        print(f"  NO INTERSECTION: '{track_feat['name']}' does NOT intersect any highway")

if not non_intersecting_ids:
    print("  All railway tracks intersect with at least one highway.")
else:

    railways.removeSelection()
    railways.select(non_intersecting_ids)
    print(f"\n  [Total: {len(non_intersecting_ids)} track(s) do not intersect any highway]")
    print("  [Non-intersecting tracks are now SELECTED/HIGHLIGHTED on the map]")

print("\n" + "=" * 65)
print("  Q2: ALL QUERIES EXECUTED SUCCESSFULLY!")
print("=" * 65)
print(\
   )
print("=" * 65)

iface.mapCanvas().refresh()
