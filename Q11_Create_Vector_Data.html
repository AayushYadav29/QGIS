road_layer = QgsVectorLayer("LineString?crs=EPSG:4326", "Roads", "memory")
road_provider = road_layer.dataProvider()
road_provider.addAttributes([
    QgsField("id", QVariant.Int),
    QgsField("name", QVariant.String),

    QgsField("length_km", QVariant.Double)

label_layers = [
    (road_layer, "name"),
    (river_layer, "name"),
    (zone_layer, "name"),
    (park_layer, "name"),
    (hospital_layer, "name"),
    (atm_layer, "bank_name")
]

for layer, field in label_layers:
    settings = QgsPalLayerSettings()
    settings.fieldName = field
    settings.enabled = True
    text_format = QgsTextFormat()
    text_format.setSize(8)
    buffer = QgsTextBufferSettings()
    buffer.setEnabled(True)
    buffer.setSize(1)
    buffer.setColor(QColor(255, 255, 255))
    text_format.setBuffer(buffer)
    settings.setFormat(text_format)
    labeling = QgsVectorLayerSimpleLabeling(settings)
    layer.setLabelsEnabled(True)
    layer.setLabeling(labeling)
    layer.triggerRepaint()

import os

download_dir = os.path.join(os.path.expanduser("~"), "Downloads")
gpkg_path = os.path.join(download_dir, "vector_data.gpkg")

print("\n" + "=" * 65)
print("  Preparing Database (GeoPackage)...")
print("=" * 65)

all_layers = {
    "Roads": road_layer,
    "Rivers": river_layer,
    "Zones": zone_layer,
    "Parks": park_layer,
    "Hospitals": hospital_layer,
    "ATMs": atm_layer
}

for i, (name, layer) in enumerate(all_layers.items()):
    options = QgsVectorFileWriter.SaveVectorOptions()
    options.driverName = "GPKG"
    options.fileEncoding = "UTF-8"
    options.layerName = name

    if i > 0:
        options.actionOnExistingFile = QgsVectorFileWriter.CreateOrOverwriteLayer

    error = QgsVectorFileWriter.writeAsVectorFormatV3(
        layer,
        gpkg_path,
        project.transformContext(),
        options
    )

    if error[0] == QgsVectorFileWriter.NoError:
        print(f"  [OK] {name:15s} -> {layer.featureCount()} features saved")
    else:
        print(f"  [FAIL] {name} -> Error: {error[1]}")

print(f"\n  Database saved: {gpkg_path}")
