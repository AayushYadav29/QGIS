from qgis.core import *
from qgis.gui import *
from PyQt5.QtGui import QColor, QFont, QImage, QPainter
from PyQt5.QtCore import QVariant, QRectF, QSizeF, Qt
import os, csv

project = QgsProject.instance()
download_dir = os.path.join(os.path.expanduser("~"), "Downloads")

project.setTitle("Mumbai City Map - GIS Practical")
print("=" * 65)
print("  Q1: Mumbai Map - Creating Layers & Layout")
print("=" * 65)

crs = QgsCoordinateReferenceSystem("EPSG:4326")
project.setCrs(crs)
print(f"\n  [Q1.1] CRS set to: {crs.authid()} ({crs.description()})")

area_layer = QgsVectorLayer("Polygon?crs=EPSG:4326", "Mumbai_Areas", "memory")
area_prov = area_layer.dataProvider()
area_prov.addAttributes([
    QgsField("id", QVariant.Int),
    QgsField("name", QVariant.String),
    QgsField("type", QVariant.String)
])
area_layer.updateFields()

areas = [
    (1, "Colaba",       "South Mumbai",
     [(72.815,18.900),(72.835,18.900),(72.835,18.920),(72.815,18.920),(72.815,18.900)]),
    (2, "Fort / CST",   "South Mumbai",
     [(72.830,18.930),(72.850,18.930),(72.850,18.950),(72.830,18.950),(72.830,18.930)]),
    (3, "Dadar",        "Central Mumbai",
     [(72.835,18.995),(72.860,18.995),(72.860,19.020),(72.835,19.020),(72.835,18.995)]),
    (4, "Bandra",       "Western Suburbs",
     [(72.815,19.050),(72.845,19.050),(72.845,19.070),(72.815,19.070),(72.815,19.050)]),
    (5, "Andheri",      "Western Suburbs",
     [(72.830,19.100),(72.870,19.100),(72.870,19.130),(72.830,19.130),(72.830,19.100)])
]

for aid, name, atype, coords in areas:
    feat = QgsFeature()
    pts = [QgsPointXY(x, y) for x, y in coords]
    feat.setGeometry(QgsGeometry.fromPolygonXY([pts]))
    feat.setAttributes([aid, name, atype])
    area_prov.addFeature(feat)

area_layer.updateExtents()
project.addMapLayer(area_layer)

symbol = QgsFillSymbol.createSimple({
    'color': '180,210,240,70',
    'outline_color': '40,80,150',
    'outline_width': '0.5'
})
area_layer.renderer().setSymbol(symbol)
area_layer.triggerRepaint()

road_layer = QgsVectorLayer("LineString?crs=EPSG:4326", "Mumbai_Roads", "memory")
road_prov = road_layer.dataProvider()
road_prov.addAttributes([
    QgsField("id", QVariant.Int),
    QgsField("name", QVariant.String)
])
road_layer.updateFields()

roads = [
    (1, "Western Express Highway", [(72.835,18.910),(72.840,18.960),(72.840,19.020),
                                     (72.835,19.060),(72.845,19.120)]),
    (2, "Eastern Express Highway",  [(72.845,18.940),(72.855,18.990),(72.860,19.050),
                                     (72.865,19.110)]),
    (3, "SV Road",                  [(72.820,19.050),(72.830,19.070),(72.835,19.110)])
]

for rid, name, coords in roads:
    feat = QgsFeature()
    pts = [QgsPointXY(x, y) for x, y in coords]
    feat.setGeometry(QgsGeometry.fromPolylineXY(pts))
    feat.setAttributes([rid, name])
    road_prov.addFeature(feat)

road_layer.updateExtents()
project.addMapLayer(road_layer)

symbol = QgsLineSymbol.createSimple({'color': '200,80,30', 'width': '0.8'})
road_layer.renderer().setSymbol(symbol)
road_layer.triggerRepaint()

rail_layer = QgsVectorLayer("LineString?crs=EPSG:4326", "Mumbai_Railways", "memory")
rail_prov = rail_layer.dataProvider()
rail_prov.addAttributes([
    QgsField("id", QVariant.Int),
    QgsField("name", QVariant.String)
])
rail_layer.updateFields()

rails = [
    (1, "Western Line",  [(72.828,18.910),(72.832,18.960),(72.833,19.020),
                           (72.830,19.065),(72.840,19.120)]),
    (2, "Central Line",  [(72.838,18.940),(72.848,18.995),(72.855,19.050),
                           (72.860,19.100)]),
    (3, "Harbour Line",  [(72.838,18.940),(72.850,18.970),(72.862,19.000)])
]

for rid, name, coords in rails:
    feat = QgsFeature()
    pts = [QgsPointXY(x, y) for x, y in coords]
    feat.setGeometry(QgsGeometry.fromPolylineXY(pts))
    feat.setAttributes([rid, name])
    rail_prov.addFeature(feat)

rail_layer.updateExtents()
project.addMapLayer(rail_layer)

symbol = QgsLineSymbol.createSimple({
    'color': '80,80,80', 'width': '0.7', 'line_style': 'dash'
})
rail_layer.renderer().setSymbol(symbol)
rail_layer.triggerRepaint()

landmark_layer = QgsVectorLayer("Point?crs=EPSG:4326", "Mumbai_Landmarks", "memory")
landmark_prov = landmark_layer.dataProvider()
landmark_prov.addAttributes([
    QgsField("id", QVariant.Int),
    QgsField("name", QVariant.String),
    QgsField("type", QVariant.String)
])
landmark_layer.updateFields()

landmarks = [
    (1, "Gateway of India",    "Monument",  72.8347, 18.9220),
    (2, "CST Station",         "Railway",   72.8353, 18.9398),
    (3, "Siddhivinayak Temple","Religious",  72.8309, 19.0170),
    (4, "Bandra-Worli Sea Link","Bridge",   72.8155, 19.0300),
    (5, "Wankhede Stadium",    "Sports",    72.8268, 18.9388),
    (6, "Haji Ali Dargah",     "Religious",  72.8128, 18.9827),
    (7, "Juhu Beach",          "Beach",     72.8267, 19.0990)
]

for lid, name, ltype, x, y in landmarks:
    feat = QgsFeature()
    feat.setGeometry(QgsGeometry.fromPointXY(QgsPointXY(x, y)))
    feat.setAttributes([lid, name, ltype])
    landmark_prov.addFeature(feat)

landmark_layer.updateExtents()
project.addMapLayer(landmark_layer)

symbol = QgsMarkerSymbol.createSimple({
    'name': 'star', 'color': '220,50,50', 'size': '4',
    'outline_color': '150,0,0', 'outline_width': '0.3'
})
landmark_layer.renderer().setSymbol(symbol)
landmark_layer.triggerRepaint()

print("\n  [Q1.6] Creating and Importing CSV File...")

csv_path = os.path.join(download_dir, "mumbai_hospitals.csv")

csv_data = [
    ["id", "name", "type", "beds", "latitude", "longitude"],
    [1, "KEM Hospital",          "Government", 1800, 18.9942, 72.8410],
    [2, "Lilavati Hospital",     "Private",    300,  19.0509, 72.8283],
    [3, "Bombay Hospital",       "Private",    750,  18.9312, 72.8282],
    [4, "Nair Hospital",         "Government", 1200, 18.9890, 72.8340],
    [5, "Hinduja Hospital",      "Private",    500,  19.0370, 72.8455],
    [6, "Sion Hospital",         "Government", 600,  19.0400, 72.8620],
    [7, "Wockhardt Hospital",    "Private",    200,  18.9660, 72.8310],
    [8, "Jaslok Hospital",       "Private",    364,  18.9710, 72.8070]
]

with open(csv_path, 'w', newline='', encoding='utf-8') as f:
    writer = csv.writer(f)
    for row in csv_data:
        writer.writerow(row)

print(f"  -> CSV file created: mumbai_hospitals.csv")

uri = f"file:///{csv_path}?delimiter=,&xField=longitude&yField=latitude&crs=EPSG:4326"
csv_layer = QgsVectorLayer(uri, "Hospitals_CSV_Import", "delimitedtext")

if csv_layer.isValid():
    project.addMapLayer(csv_layer)
    symbol = QgsMarkerSymbol.createSimple({
        'name': 'cross', 'color': '0,150,0', 'size': '4.5',
        'outline_color': '0,100,0', 'outline_width': '0.5'
    })
    csv_layer.renderer().setSymbol(symbol)
    csv_layer.triggerRepaint()
    print(f"  -> CSV imported as layer: {csv_layer.name()} ({csv_layer.featureCount()} features)")
else:
    print("  -> ERROR: Could not import CSV layer!")

print("\n  [Q1.7] Layer Names and Geometry Info (CRS: WGS 84):")
print(f"  {'Layer Name':<25s} {'Geometry':<12s} {'Features':>8s}  {'CRS'}")
print(f"  {'-'*25} {'-'*12} {'-'*8}  {'-'*15}")

all_layers = [area_layer, road_layer, rail_layer, landmark_layer]
if csv_layer.isValid():
    all_layers.append(csv_layer)

for layer in all_layers:
    geom_type = QgsWkbTypes.displayString(layer.wkbType())
    print(f"  {layer.name():<25s} {geom_type:<12s} {layer.featureCount():>8d}  {layer.crs().authid()}")

label_configs = [
    (area_layer, "name"),
    (road_layer, "name"),
    (rail_layer, "name"),
    (landmark_layer, "name"),
]
if csv_layer.isValid():
    label_configs.append((csv_layer, "name"))

for layer, field in label_configs:
    settings = QgsPalLayerSettings()
    settings.fieldName = field
    settings.enabled = True
    text_format = QgsTextFormat()
    text_format.setSize(8)
    buf = QgsTextBufferSettings()
    buf.setEnabled(True)
    buf.setSize(1)
    buf.setColor(QColor(255, 255, 255))
    text_format.setBuffer(buf)
    settings.setFormat(text_format)
    labeling = QgsVectorLayerSimpleLabeling(settings)
    layer.setLabelsEnabled(True)
    layer.setLabeling(labeling)
    layer.triggerRepaint()

canvas = iface.mapCanvas()

center = QgsPointXY(72.845, 19.00)
canvas.setCenter(center)
canvas.zoomScale(40000)
canvas.refresh()
print(f"\n  [Q1.1] Scale set to 1:{int(canvas.scale())}")

print("\n  [Q1.4] Setting project properties...")
project.setTitle("Mumbai City Map")
project.writeEntry("WMSServiceTitle", "/", "Mumbai City Map - GIS Practical")
project.setBackgroundColor(QColor(240, 248, 255))
print("  -> Project title: Mumbai City Map")
print("  -> Background color: Light blue (Alice Blue)")

print("\n  Creating Print Layout with cartographic elements...")

layout_name = "Mumbai_Map_Layout"

manager = project.layoutManager()
existing = manager.layoutByName(layout_name)
if existing:
    manager.removeLayout(existing)

layout = QgsPrintLayout(project)
layout.initializeDefaults()
layout.setName(layout_name)
manager.addLayout(layout)

map_item = QgsLayoutItemMap(layout)
map_item.setRect(20, 20, 200, 250)
map_item.setExtent(canvas.extent())
map_item.setCrs(crs)
layout.addLayoutItem(map_item)
map_item.attemptMove(QgsLayoutPoint(15, 30, QgsUnitTypes.LayoutMillimeters))
map_item.attemptResize(QgsLayoutSize(180, 220, QgsUnitTypes.LayoutMillimeters))

map_item.setScale(40000)
print(f"  -> Map added to layout (Scale 1:40000)")

title = QgsLayoutItemLabel(layout)
title.setText("MUMBAI CITY MAP")
title.setFont(QFont("Arial", 20, QFont.Bold))
title.adjustSizeToText()
layout.addLayoutItem(title)
title.attemptMove(QgsLayoutPoint(50, 5, QgsUnitTypes.LayoutMillimeters))
print("  [Q1.5] -> Title added: MUMBAI CITY MAP")

legend = QgsLayoutItemLegend(layout)
legend.setTitle("Legend")
legend.setLinkedMap(map_item)
layout.addLayoutItem(legend)
legend.attemptMove(QgsLayoutPoint(200, 30, QgsUnitTypes.LayoutMillimeters))
print("  [Q1.2] -> Legend added")

scalebar = QgsLayoutItemScaleBar(layout)
scalebar.setStyle('Single Box')
scalebar.setLinkedMap(map_item)
scalebar.setMapUnitsPerScaleBarUnit(1000)
scalebar.setNumberOfSegments(4)
scalebar.setNumberOfSegmentsLeft(0)
scalebar.setUnitsPerSegment(1)
scalebar.setUnitLabel("km")
layout.addLayoutItem(scalebar)
scalebar.attemptMove(QgsLayoutPoint(15, 255, QgsUnitTypes.LayoutMillimeters))
print("  [Q1.3] -> Scale Bar added")

north = QgsLayoutItemLabel(layout)
north.setText("N\n▲")
north.setFont(QFont("Arial", 16, QFont.Bold))
north.adjustSizeToText()
layout.addLayoutItem(north)
north.attemptMove(QgsLayoutPoint(200, 240, QgsUnitTypes.LayoutMillimeters))
print("  -> North Arrow added")

output_path = os.path.join(download_dir, "Mumbai_Map_Output.jpg")

exporter = QgsLayoutExporter(layout)
settings = QgsLayoutExporter.ImageExportSettings()
settings.dpi = 300

result = exporter.exportToImage(output_path, settings)

if result == QgsLayoutExporter.Success:
    print(f"\n  -> Map exported: Mumbai_Map_Output.jpg (300 dpi)")
else:
    print(f"\n  -> Export status code: {result}")

print("\n" + "=" * 65)
print("  Q1: Mumbai Map - COMPLETED SUCCESSFULLY!")
print("=" * 65)

print("  2. Legend        : Added to print layout")

print("  5. Map Name      : 'MUMBAI CITY MAP'")
print("  6. CSV Import    : mumbai_hospitals.csv (8 hospitals)")
print("  7. Layer Info    : Names & geometry types with CRS shown above")
print(f"\n  Output: Mumbai_Map_Output.jpg")
print(f"  Layout: {layout_name} (View via Project > Layouts)")
print("=" * 65)
