from qgis.core import *
from qgis.gui import *
from PyQt5.QtGui import QColor
from PyQt5.QtCore import QVariant
import processing
import os

project = QgsProject.instance()
download_dir = os.path.join(os.path.expanduser("~"), "Downloads")

res_list     = project.mapLayersByName("Residential_Areas")
shops_list   = project.mapLayersByName("Shops")
schools_list = project.mapLayersByName("Schools")
clinics_list = project.mapLayersByName("Clinics")

if not res_list or not shops_list or not schools_list or not clinics_list:
    print("ERROR: Please run Q13_Residential_Join.py first!")
    raise Exception("Required layers not found. Run Q13 first.")

residential = res_list[0]
shops       = shops_list[0]
schools     = schools_list[0]
clinics     = clinics_list[0]

print("=" * 65)
print("  Q2: Count Points in Polygon - Analysis Tool Results")
print("=" * 65)

print("\n" + "-" * 65)
print("  Analysis 1: Count SHOPS in each Residential Area")
print("-" * 65)

shops_output = os.path.join(download_dir, "count_shops_in_residential.shp")

try:
    result = processing.run("native:countpointsinpolygon", {
        'POLYGONS': residential,
        'POINTS': shops,
        'WEIGHT': '',
        'CLASSFIELD': '',
        'FIELD': 'SHOP_COUNT',
        'OUTPUT': shops_output
    })

    shops_result = QgsVectorLayer(shops_output, "Shops_Count_Per_Area", "ogr")
    if shops_result.isValid():
        project.addMapLayer(shops_result)

        ranges = []
        sym0 = QgsFillSymbol.createSimple({'color':'220,240,220,120','outline_color':'0,100,0','outline_width':'0.4'})
        sym1 = QgsFillSymbol.createSimple({'color':'144,238,144,150','outline_color':'0,100,0','outline_width':'0.4'})
        sym2 = QgsFillSymbol.createSimple({'color':'60,179,113,180','outline_color':'0,80,0','outline_width':'0.4'})
        ranges.append(QgsRendererRange(0, 1, sym0, "0-1 shops"))
        ranges.append(QgsRendererRange(1, 2, sym1, "1-2 shops"))
        ranges.append(QgsRendererRange(2, 5, sym2, "2+ shops"))
        renderer = QgsGraduatedSymbolRenderer("SHOP_COUNT", ranges)
        shops_result.setRenderer(renderer)
        shops_result.triggerRepaint()

        print(f"\n  {'Residential Area':<22s} {'Shop Count':>12s}")
        print(f"  {'-'*22} {'-'*12}")
        for feat in shops_result.getFeatures():
            print(f"  {feat['name']:<22s} {feat['SHOP_COUNT']:>12d}")

        print(f"\n  [OK] Output: count_shops_in_residential.shp")
        print("  [OK] Graduated color applied (lighter = fewer shops)")

except Exception as e:
    print(f"  [NOTE] Processing: {e}")

print("\n" + "-" * 65)
print("  Analysis 2: Count SCHOOLS in each Residential Area")
print("-" * 65)

schools_output = os.path.join(download_dir, "count_schools_in_residential.shp")

try:
    result = processing.run("native:countpointsinpolygon", {
        'POLYGONS': residential,
        'POINTS': schools,
        'WEIGHT': '',
        'CLASSFIELD': '',
        'FIELD': 'SCHOOL_CNT',
        'OUTPUT': schools_output
    })

    schools_result = QgsVectorLayer(schools_output, "Schools_Count_Per_Area", "ogr")
    if schools_result.isValid():
        project.addMapLayer(schools_result)

        print(f"\n  {'Residential Area':<22s} {'School Count':>12s}")
        print(f"  {'-'*22} {'-'*12}")
        for feat in schools_result.getFeatures():
            print(f"  {feat['name']:<22s} {feat['SCHOOL_CNT']:>12d}")

        print(f"\n  [OK] Output: count_schools_in_residential.shp")

except Exception as e:
    print(f"  [NOTE] Processing: {e}")

print("\n" + "-" * 65)
print("  Analysis 3: Count CLINICS in each Residential Area")
print("-" * 65)

clinics_output = os.path.join(download_dir, "count_clinics_in_residential.shp")

try:
    result = processing.run("native:countpointsinpolygon", {
        'POLYGONS': residential,
        'POINTS': clinics,
        'WEIGHT': '',
        'CLASSFIELD': '',
        'FIELD': 'CLINIC_CNT',
        'OUTPUT': clinics_output
    })

    clinics_result = QgsVectorLayer(clinics_output, "Clinics_Count_Per_Area", "ogr")
    if clinics_result.isValid():
        project.addMapLayer(clinics_result)

        print(f"\n  {'Residential Area':<22s} {'Clinic Count':>12s}")
        print(f"  {'-'*22} {'-'*12}")
        for feat in clinics_result.getFeatures():
            print(f"  {feat['name']:<22s} {feat['CLINIC_CNT']:>12d}")

        print(f"\n  [OK] Output: count_clinics_in_residential.shp")

except Exception as e:
    print(f"  [NOTE] Processing: {e}")

print("\n" + "-" * 65)
print("  COMBINED SUMMARY (Manual Verification)")
print("-" * 65)

shop_features   = list(shops.getFeatures())
school_features = list(schools.getFeatures())
clinic_features = list(clinics.getFeatures())

print(f"\n  {'Residential Area':<22s} {'Shops':>7s} {'Schools':>9s} {'Clinics':>9s} {'Total':>7s}")
print(f"  {'-'*22} {'-'*7} {'-'*9} {'-'*9} {'-'*7}")

grand_shops = grand_schools = grand_clinics = 0

for res_feat in residential.getFeatures():
    res_geom = res_feat.geometry()
    res_name = res_feat['name']

    s_count = sum(1 for s in shop_features if res_geom.contains(s.geometry()))
    sc_count = sum(1 for s in school_features if res_geom.contains(s.geometry()))
    c_count = sum(1 for c in clinic_features if res_geom.contains(c.geometry()))
    total = s_count + sc_count + c_count

    grand_shops += s_count
    grand_schools += sc_count
    grand_clinics += c_count

    print(f"  {res_name:<22s} {s_count:>7d} {sc_count:>9d} {c_count:>9d} {total:>7d}")

print(f"  {'='*22} {'='*7} {'='*9} {'='*9} {'='*7}")
print(f"  {'TOTAL':<22s} {grand_shops:>7d} {grand_schools:>9d} {grand_clinics:>9d} {grand_shops+grand_schools+grand_clinics:>7d}")

print("\n" + "=" * 65)
print("  Q2: Count Points in Polygon - COMPLETED!")
print("=" * 65)
print("  Analysis Tool Used: native:countpointsinpolygon")
print("  Polygon Layer: Residential_Areas (6 areas)")
print("  Point Layers Counted:")
print(f"    - Shops   : {shops.featureCount()} points -> SHOP_COUNT field")
print(f"    - Schools : {schools.featureCount()} points -> SCHOOL_CNT field")
print(f"    - Clinics : {clinics.featureCount()} points -> CLINIC_CNT field")
print("  Outputs:")
print("    - count_shops_in_residential.shp (with graduated colors)")
print("    - count_schools_in_residential.shp")
print("    - count_clinics_in_residential.shp")
print("=" * 65)
