from qgis.core import *
from qgis.gui import *
from PyQt5.QtCore import QVariant
import os

project = QgsProject.instance()

layer_names = [
    "Railway_Track",
    "Railway_Stations",
    "Adjacent_Buildings",
    "Roads",
    "Localities",
    "Landmarks",
    "Water_Bodies"
]

layers = {}
missing = []
for name in layer_names:
    found = project.mapLayersByName(name)
    if found:
        layers[name] = found[0]
    else:
        missing.append(name)

if missing:
    print(f"ERROR: Missing layers: {', '.join(missing)}")
    print("Please run Q5_Railway_CST_Chembur_Map.py first!")
    raise Exception("Required layers not found. Run Q5 first.")

print("=" * 65)
print("  Q2: Creating Datasets for All Layers")
print("=" * 65)

download_dir = os.path.join(os.path.expanduser("~"), "Downloads")
dataset_dir = os.path.join(download_dir, "datasets")

if not os.path.exists(dataset_dir):
    os.makedirs(dataset_dir)
    print(f"\n  Created directory: {dataset_dir}")
else:
    print(f"\n  Output directory: {dataset_dir}")

print("\n" + "-" * 65)
print("  Exporting Layers as Shapefiles (.shp)")
print("-" * 65)

for name, layer in layers.items():
    shp_path = os.path.join(dataset_dir, f"{name}.shp")

    options = QgsVectorFileWriter.SaveVectorOptions()
    options.driverName = "ESRI Shapefile"
    options.fileEncoding = "UTF-8"

    error = QgsVectorFileWriter.writeAsVectorFormatV3(
        layer,
        shp_path,
        project.transformContext(),
        options
    )

    if error[0] == QgsVectorFileWriter.NoError:
        print(f"  [OK] {name}.shp  ({layer.featureCount()} features)")
    else:
        print(f"  [FAIL] {name}.shp -> Error: {error[1]}")

print("\n" + "-" * 65)
print("  Exporting Layers as CSV")
print("-" * 65)

for name, layer in layers.items():
    csv_path = os.path.join(dataset_dir, f"{name}.csv")

    options = QgsVectorFileWriter.SaveVectorOptions()
    options.driverName = "CSV"
    options.fileEncoding = "UTF-8"

    options.layerOptions = ["GEOMETRY=AS_WKT"]

    error = QgsVectorFileWriter.writeAsVectorFormatV3(
        layer,
        csv_path,
        project.transformContext(),
        options
    )

    if error[0] == QgsVectorFileWriter.NoError:
        print(f"  [OK] {name}.csv  ({layer.featureCount()} records)")
    else:
        print(f"  [FAIL] {name}.csv -> Error: {error[1]}")

print("\n" + "-" * 65)
print("  Saving QGIS Project File")
print("-" * 65)

project_path = os.path.join(dataset_dir, "CST_Chembur_Railway.qgz")
project.write(project_path)
print(f"  [OK] Project saved: CST_Chembur_Railway.qgz")

print("\n" + "=" * 65)
print("  Q2: Dataset Creation - COMPLETED SUCCESSFULLY!")
print("=" * 65)
print(f"\n  Output Location: {dataset_dir}")
print(f"\n  Files Created:")
print(f"  {'='*50}")

for name in layer_names:
    print(f"  Shapefile : {name}.shp (.shx, .dbf, .prj)")
    print(f"  CSV       : {name}.csv")

print(f"  Project   : CST_Chembur_Railway.qgz")
print(f"\n  Total Layers Exported: {len(layers)}")
print(f"  Formats: Shapefile (.shp) + CSV + QGIS Project (.qgz)")
print("=" * 65)
