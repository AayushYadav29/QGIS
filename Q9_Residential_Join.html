for layer, field in [(res_layer,"name"),(shop_layer,"name"),(school_layer,"name"),(clinic_layer,"name")]:
    settings = QgsPalLayerSettings()
    settings.fieldName = field
    settings.enabled = True
    tf = QgsTextFormat()
    tf.setSize(7)
    buf = QgsTextBufferSettings()
    buf.setEnabled(True)
    buf.setSize(1)
    buf.setColor(QColor(255,255,255))
    tf.setBuffer(buf)
    settings.setFormat(tf)
    layer.setLabelsEnabled(True)
    layer.setLabeling(QgsVectorLayerSimpleLabeling(settings))
    layer.triggerRepaint()

print("\n" + "-" * 65)
print("  SPATIAL JOIN: Join Shop data to Residential Areas by Location")
print("-" * 65)

join_output = os.path.join(download_dir, "spatial_join_result.shp")

try:
    result = processing.run("native:joinattributesbylocation", {
        'INPUT': res_layer,
        'JOIN': shop_layer,
        'PREDICATE': [0],
        'JOIN_FIELDS': ['name', 'shop_type', 'monthly_revenue'],
        'METHOD': 0,
        'DISCARD_NONMATCHING': False,
        'PREFIX': 'shop_',
        'OUTPUT': join_output
    })

    joined_layer = QgsVectorLayer(join_output, "Spatial_Join_Result", "ogr")
    if joined_layer.isValid():
        project.addMapLayer(joined_layer)

        sym = QgsFillSymbol.createSimple({
            'color': '180,255,180,80',
            'outline_color': '0,120,0',
            'outline_width': '0.5'
        })
        joined_layer.renderer().setSymbol(sym)
        joined_layer.triggerRepaint()

        print("\n  Spatial Join Results (Residential Area -> Shops inside):")
        print(f"  {'Residential Area':<22s} {'Shop Name':<20s} {'Type':<12s} {'Revenue':>10s}")
        print(f"  {'-'*22} {'-'*20} {'-'*12} {'-'*10}")

        for feat in joined_layer.getFeatures():
            res_name = feat['name']
            shop_name = feat['shop_name'] if feat['shop_name'] else "(none)"
            shop_type = feat['shop_type'] if feat['shop_type'] else ""
            revenue = feat['shop_monthly_revenue'] if feat['shop_monthly_revenue'] else 0
            print(f"  {res_name:<22s} {shop_name:<20s} {shop_type:<12s} {revenue:>10.0f}")

        print(f"\n  [OK] Spatial Join output: spatial_join_result.shp")
    else:
        print("  [NOTE] Could not load joined layer")

except Exception as e:
    print(f"  [NOTE] Processing: {e}")

print("\n  --- Manual Spatial Join Verification ---")
print(f"\n  {'Residential Area':<22s} {'Shops Inside':>15s}")
print(f"  {'-'*22} {'-'*15}")

shop_features = list(shop_layer.getFeatures())

for res_feat in res_layer.getFeatures():
    res_geom = res_feat.geometry()
    res_name = res_feat['name']
    shops_inside = []
    for s in shop_features:
        if res_geom.contains(s.geometry()):
            shops_inside.append(s['name'])

    print(f"  {res_name:<22s} {len(shops_inside):>15d}")
    for sname in shops_inside:
        print(f"    -> {sname}")
