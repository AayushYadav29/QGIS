from qgis.core import *
from qgis.gui import *
from PyQt5.QtGui import QColor
from PyQt5.QtCore import QVariant
import processing
import os

project = QgsProject.instance()

hospitals_list = project.mapLayersByName("Hospitals")
atms_list      = project.mapLayersByName("ATMs")
zones_list     = project.mapLayersByName("Zones")
parks_list     = project.mapLayersByName("Parks")

if not hospitals_list or not atms_list or not zones_list or not parks_list:
    print("ERROR: Please run Q7_Create_Vector_Data.py first to create all layers!")
    raise Exception("Required layers not found. Run Q7 first.")

hospitals = hospitals_list[0]
atms      = atms_list[0]
zones     = zones_list[0]
parks     = parks_list[0]

download_dir = os.path.join(os.path.expanduser("~"), "Downloads")

print("=" * 65)
print("  Q2: Analytical Tools Demonstration - RESULTS")
print("=" * 65)

print("\n" + "-" * 65)
print("  TOOL 1: Distance Matrix (Hospitals -> ATMs)")
print("-" * 65)
print("  Computing distance from each Hospital to nearest ATMs...\n")

output_csv = os.path.join(download_dir, "distance_matrix_output.csv")

try:
    result = processing.run("qgis:distancematrix", {
        'INPUT': hospitals,
        'INPUT_FIELD': 'name',
        'TARGET': atms,
        'TARGET_FIELD': 'bank_name',
        'MATRIX_TYPE': 0,
        'NEAREST_POINTS': 3,
        'OUTPUT': output_csv
    })
    print(f"  [OK] Distance matrix saved: distance_matrix_output.csv")

    dm_layer = QgsVectorLayer(output_csv, "Distance_Matrix_Result", "ogr")
    if dm_layer.isValid():
        project.addMapLayer(dm_layer)

        for feat in dm_layer.getFeatures():
            fields = feat.fields()
            values = feat.attributes()
            input_name = values[0]
            target_name = values[1]
            distance = values[2]
            print(f"  {input_name:30s} -> {target_name:15s} : {distance:10.2f} meters")
    else:
        print("  [INFO] Result layer not loadable, see CSV file for data.")

except Exception as e:
    print(f"  [NOTE] Processing algorithm returned: {e}")
    print("  Calculating distances manually...\n")

print("\n  --- Manual Distance Calculation ---")

d = QgsDistanceArea()
d.setEllipsoid('WGS84')
d.setSourceCrs(hospitals.crs(), project.transformContext())

hospital_features = list(hospitals.getFeatures())
atm_features = list(atms.getFeatures())

for h_feat in hospital_features:
    h_point = h_feat.geometry().asPoint()
    h_name = h_feat['name']

    distances = []
    for a_feat in atm_features:
        a_point = a_feat.geometry().asPoint()
        a_name = a_feat['bank_name']
        dist = d.measureLine(QgsPointXY(h_point), QgsPointXY(a_point))
        distances.append((a_name, dist))

    distances.sort(key=lambda x: x[1])

    print(f"\n  Hospital: {h_name}")
    print(f"  {'ATM':<20s} {'Distance (m)':>15s}")
    print(f"  {'-'*20} {'-'*15}")
    for atm_name, dist in distances[:3]:
        print(f"  {atm_name:<20s} {dist:>15.2f}")

print("\n\n" + "-" * 65)
print("  TOOL 2: Points in Polygon (Count Points in Zones)")
print("-" * 65)

print("\n  A) Hospitals in each Zone:")

try:
    hosp_output = os.path.join(download_dir, "hospitals_in_zones.shp")
    result = processing.run("native:countpointsinpolygon", {
        'POLYGONS': zones,
        'POINTS': hospitals,
        'WEIGHT': '',
        'CLASSFIELD': '',
        'FIELD': 'HOSP_COUNT',
        'OUTPUT': hosp_output
    })

    hosp_result_layer = QgsVectorLayer(hosp_output, "Hospitals_in_Zones", "ogr")
    if hosp_result_layer.isValid():
        project.addMapLayer(hosp_result_layer)
        for feat in hosp_result_layer.getFeatures():
            print(f"  Zone: {feat['name']:20s} -> Hospitals: {feat['HOSP_COUNT']}")
    print(f"  [OK] Output saved: hospitals_in_zones.shp")

except Exception as e:
    print(f"  [NOTE] Processing returned: {e}")

print("\n  B) ATMs in each Zone:")

try:
    atm_output = os.path.join(download_dir, "atms_in_zones.shp")
    result = processing.run("native:countpointsinpolygon", {
        'POLYGONS': zones,
        'POINTS': atms,
        'WEIGHT': '',
        'CLASSFIELD': '',
        'FIELD': 'ATM_COUNT',
        'OUTPUT': atm_output
    })

    atm_result_layer = QgsVectorLayer(atm_output, "ATMs_in_Zones", "ogr")
    if atm_result_layer.isValid():
        project.addMapLayer(atm_result_layer)
        for feat in atm_result_layer.getFeatures():
            print(f"  Zone: {feat['name']:20s} -> ATMs: {feat['ATM_COUNT']}")
    print(f"  [OK] Output saved: atms_in_zones.shp")

except Exception as e:
    print(f"  [NOTE] Processing returned: {e}")

print("\n  --- Manual Points-in-Polygon Count ---")

print(f"\n  {'Zone':<20s} {'Hospitals':>10s} {'ATMs':>10s}")
print(f"  {'-'*20} {'-'*10} {'-'*10}")

for zone_feat in zones.getFeatures():
    zone_geom = zone_feat.geometry()
    zone_name = zone_feat['name']

    hosp_count = 0
    for h_feat in hospital_features:
        if zone_geom.contains(h_feat.geometry()):
            hosp_count += 1

    atm_count = 0
    for a_feat in atm_features:
        if zone_geom.contains(a_feat.geometry()):
            atm_count += 1

    print(f"  {zone_name:<20s} {hosp_count:>10d} {atm_count:>10d}")

print(f"\n  {'Park':<20s} {'Hospitals':>10s} {'ATMs':>10s}")
print(f"  {'-'*20} {'-'*10} {'-'*10}")

for park_feat in parks.getFeatures():
    park_geom = park_feat.geometry()
    park_name = park_feat['name']

    hosp_count = sum(1 for h in hospital_features if park_geom.contains(h.geometry()))
    atm_count = sum(1 for a in atm_features if park_geom.contains(a.geometry()))

    print(f"  {park_name:<20s} {hosp_count:>10d} {atm_count:>10d}")

print("\n" + "=" * 65)
print("  Q2: Analytical Tools Demonstration - COMPLETED!")
print("=" * 65)
print("  Tool 1: Distance Matrix")
print("    -> Calculated distance from each Hospital to nearest 3 ATMs")
print("    -> Output: distance_matrix_output.csv")
print("  Tool 2: Points in Polygon")
print("    -> Counted Hospitals & ATMs in each Zone and Park")
print("    -> Output: hospitals_in_zones.shp, atms_in_zones.shp")
print("=" * 65)
